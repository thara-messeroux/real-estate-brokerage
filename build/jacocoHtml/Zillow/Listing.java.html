package Zillow;

import java.util.Objects;

/**
 * represent a generic Listing containing a contract and property
 * @param <P> - AbstractProperty, a property class object
 * @param <C> - Contract, a contract class object
 */
public class Listing<P extends AbstractProperty, C extends AbstractContract > {

  private P property;
  private C contract;
  /**
   * Constructor for a Listing
   * @param property - AbstractProperty, represents listing's property
   * @param contract - Contract, represent listing's contract

   */
  public Listing(P property, C contract ) {
    this.property = property;
    this.contract = contract;
  }

  /**
   * @return - Property
   */
  public P getProperty() {
    return this.property;
  }

  /**
   * @return - Contract,
   */
  public C getContract() {
    return this.contract;
  }

  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (o == null || getClass() != o.getClass()) {
      return false;
    }
    Listing<?, ?> listing = (Listing<?, ?>) o;
    return property.equals(listing.property) && contract.equals(listing.contract);
  }

  @Override
  public int hashCode() {
    return Objects.hash(property, contract);
  }

  @Override
  public String toString() {
    return "Listing{" +
        "property=" + property +
        ", contract=" + contract +
        '}';
  }
}
